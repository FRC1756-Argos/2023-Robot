57constexpr std::optional<SetpointPosition> GetTargetPosition(ScoringPosition gridPosition,
58 bool enableBashGuard,
59 WristPosition wristPosition) {
60 bool wristInverted = wristPosition == WristPosition::RollersDown;
61 SetpointPosition targetPosition;
62 if (gridPosition.column == ScoringColumn::coneIntake) {
63 targetPosition = scoring_positions::lifter_extension_end::coneIntake;
64 } else if (gridPosition.column == ScoringColumn::cubeIntake) {
65 targetPosition = scoring_positions::lifter_extension_end::cubeIntake;
66 } else if (gridPosition.column == ScoringColumn::stow) {
67 targetPosition = scoring_positions::lifter_extension_end::stow;
68 } else if (gridPosition.column == ScoringColumn::leftGrid_leftCone ||
69 gridPosition.column == ScoringColumn::leftGrid_rightCone ||
70 gridPosition.column == ScoringColumn::middleGrid_leftCone ||
71 gridPosition.column == ScoringColumn::middleGrid_rightCone ||
72 gridPosition.column == ScoringColumn::rightGrid_leftCone ||
73 gridPosition.column == ScoringColumn::rightGrid_rightCone) {
74 if (gridPosition.row == ScoringRow::low) {
75 targetPosition = wristInverted ? scoring_positions::lifter_extension_end::coneLow_wristInverted :
76 scoring_positions::lifter_extension_end::coneLow;
77 } else if (gridPosition.row == ScoringRow::middle) {
78 targetPosition = wristInverted ? scoring_positions::lifter_extension_end::coneMid_wristInverted :
79 scoring_positions::lifter_extension_end::coneMid;
80 } else if (gridPosition.row == ScoringRow::high) {
81 targetPosition = wristInverted ? scoring_positions::lifter_extension_end::coneHigh_wristInverted :
82 scoring_positions::lifter_extension_end::coneHigh;
83 } else {
84 return std::nullopt;
85 }
86 if (gridPosition.column == ScoringColumn::rightGrid_leftCone) {
87 targetPosition.lifterPosition =
88 scoring_positions::lifter_extension_end::floppyConeOffset + targetPosition.lifterPosition;
89 }
90 } else if (gridPosition.column == ScoringColumn::leftGrid_middleCube ||
91 gridPosition.column == ScoringColumn::middleGrid_middleCube ||
92 gridPosition.column == ScoringColumn::rightGrid_middleCube) {
93 if (gridPosition.row == ScoringRow::low) {
94 targetPosition = wristInverted ? scoring_positions::lifter_extension_end::cubeLow_wristInverted :
95 scoring_positions::lifter_extension_end::cubeLow;
96 } else if (gridPosition.row == ScoringRow::middle) {
97 targetPosition = wristInverted ? scoring_positions::lifter_extension_end::cubeMid_wristInverted :
98 scoring_positions::lifter_extension_end::cubeMid;
99 } else if (gridPosition.row == ScoringRow::high) {
100 targetPosition = wristInverted ? scoring_positions::lifter_extension_end::cubeHigh_wristInverted :
101 scoring_positions::lifter_extension_end::cubeHigh;
102 } else {
103 return std::nullopt;
104 }
105 } else {
106 return std::nullopt;
107 }
108
109 return SetpointPosition{targetPosition.lifterPosition,
110 enableBashGuard ? targetPosition.bashGuardPosition : BashGuardPosition::Stationary};
111}
